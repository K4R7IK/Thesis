% Below content is just for example and based on my research paper;
% You can modify it according to your needs for specific languages
% Define YAML as a custom language
\lstdefinelanguage{yaml}{
	morekeywords={true,false,null,y,n,yes,no},
	sensitive=true,
	morecomment=[l]{\#},
	morestring=[b]",
}

% Default style for YAML listings

	keywordstyle=\color{blue},
	commentstyle=\color{gray},
	stringstyle=\color{red},
	numbers=left,
	numbersep=5pt,
	breaklines=true,
	frame=single
}
